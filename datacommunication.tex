       \textbf{2. What is the difference between pure and slotted Aloha?  Derive the efficiency of them.
   \vskip 12pt
   Ans:-}
   \vskip 2em
    \begin{tabular}{|m{12em}|m{12em}|}
    \hline
    \bf Pure Aloha & \bf Slotted Aloha\\
    \hline \footnotesize \textbf{1.} In pure Aloha, data can transmit at any time in any station. & \footnotesize \textbf{1.} In Slotted Aloha data can transmit only at the beginning of a time slot in any station\\
    \hline \footnotesize \textbf{2.} Time is continuous in pure aloha and it is not globally synchronized. & \footnotesize \textbf{2.} Time is discrete in slotted aloha and it is globally synchronized.\\
    \hline \footnotesize \textbf{3.} The vulnerable time or susceptible time in pure aloha is equal to $(2 \times Tt)$. & \footnotesize \textbf{3.} The vulnerabletime in slotted aloha is equal to $(Tt)$.\\
    \hline \footnotesize \textbf{4.} The maximum efficiency is equal to 18.4\%. & \footnotesize \textbf{4.} The maximum efficiency is equal to 36.8\%.\\
    \hline \footnotesize \textbf{5.} The probability of successfull transmission of a data packet $ S = G\times e^{-2G}$. & \footnotesize \textbf{5.} The probability of successfull transmission of data packet $S = G\times e^{-G}$.\\
    \hline \footnotesize \textbf{6.} It does not reduce the number of collisions. & \footnotesize \textbf{6.} It reduces the number of collisions to half, thus double the efficiency.\\
    \hline
    \end{tabular}

\underline{Derivation the efficiency of Pure Aloha}\\
\par Here,\\
\begin{equation}
 S=G\times e^{-2G}
\end{equation}
 For Maximum efficiency\\
 \vskip 4pt
$\Rightarrow \frac{dS}{dG}=0$ \\
\par $\Rightarrow \frac{d(G\times e^{-2G})}{dG}=0$\\
\par $\Rightarrow \frac{Gd(e^{-2G})}{dG}+\frac{e^{-2G}dG}{dG}=0$\\
\par $\Rightarrow -2Ge^{-2G}+e^{-2G}=0$\\     
\par $\Rightarrow e^{-2G}(1-2G)=0$\\\\
$\therefore$ $ G=\frac{1}{2}$\\\\
 From equation(1), putting $G=\frac{1}{2}$\\\\
\par $ S=\frac{1}{2}\times e^{-2\times \frac{1}{2}}$\\
\par $=\frac{1}{2} \times e^{-1}$\\
\par $=\frac{1}{2}\times \frac{1}{e}$\\
\par $=\frac{1}{2}\times \frac{1}{2.718}$\\\\
$\therefore$ $S=18.4\%$
\newpage
\underline{Derivation the efficiency of Slotted Aloha}\\
\par Here,\\
\begin{equation}
  S=G\times e^{-G}\\
\end{equation}
For Maximum efficiency \\
\vskip 4pt
\par $\Rightarrow \frac{dS}{dG}=0$\\ 
\par $\Rightarrow \frac{d(G\times e^{-G})}{dG}=0$\\
\par $\Rightarrow \frac{Gd(e^{-G})}{dG}+\frac{e^{-G}dG}{dG}=0$\\
\par $\Rightarrow -Ge^{-G}+e^{-G}=0$\\
\par $\Rightarrow e^{-G}(1-G)=0$\\\\
$\therefore$ $G=1$\\\\
From equation(2), putting G=1\\
\par $S=1\times e^{-1\times 1}$\\
\par $=1 \times e^{-1}$\\
\par $=1\times \frac{1}{e}$\\
\par $=1\times \frac{1}{2.718}$\\\\
$\therefore$ $=36.8\%$
